LinkedList and JUnit Practice

Create LinkedList class with inner class Node

public class LinkedList {
      private class Node {
        int value;
        Node next; // reference to “another” Node
        // It simply reserves space for a reference, which defaults to null
      
        Node(int value, Node next){
          this.value = value;
          this.next = next;
        }
      }

      private Node head; //start of the list
	
	public LinkedList() {
		head = null;// empty list
	}

}

Add method to add node at the beginning

public void addAtBeginning(int value) {
      this.head = new Node(value, head); 
    }

Add method to calculate size

public int size() {
        int count = 0;
        Node current = this.head;
        while(current!=null) {
          count++;
          current = current.next;
        }
        return count;
      }

Add method to insert a value after a specific value

public void addAfterValue(int valueToAdd, int valueToInsertAfter) {
          Node current = this.head;
          
          while(current.value!=valueToInsertAfter) {
            current = current.next;
          }
          current.next = new Node(valueToAdd, current.next);
        }

Add override toString()


            public String toString() {
              Node current = this.head;
              String result ="";
              while(current!=null) {
                result+= current.value;
                result+="->";
                current=current.next;
              }
              result+="null";
              return result;
            }
